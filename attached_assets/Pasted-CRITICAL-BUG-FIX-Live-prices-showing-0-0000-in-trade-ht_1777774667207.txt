METHOD 1 — Use allorigins CORS proxy for Yahoo Finance:

async function fetchPrice(symbol, isASX = false) {
  const ticker = isASX ? symbol + '.AX' : symbol;
  const url = `https://api.allorigins.win/raw?url=${encodeURIComponent(
    `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?interval=1d&range=1d`
  )}`;
  
  try {
    const res = await fetch(url);
    const data = await res.json();
    const quote = data.chart.result[0].meta;
    return {
      price: quote.regularMarketPrice,
      change: quote.regularMarketChangePercent,
      high: quote.regularMarketDayHigh,
      low: quote.regularMarketDayLow,
      volume: quote.regularMarketVolume,
      prevClose: quote.chartPreviousClose
    };
  } catch(e) {
    return null;
  }
}

async function fetchCrypto() {
  const url = 'https://api.coingecko.com/api/v3/simple/price?ids=bitcoin,ethereum,solana&vs_currencies=aud&include_24hr_change=true&include_24hr_vol=true&include_high_24h=true&include_low_24h=true';
  const res = await fetch(url);
  const data = await res.json();
  return {
    BTC: { price: data.bitcoin.aud, change: data.bitcoin.aud_24h_change },
    ETH: { price: data.ethereum.aud, change: data.ethereum.aud_24h_change },
    SOL: { price: data.solana.aud, change: data.solana.aud_24h_change }
  };
}

MASTER FETCH FUNCTION:
async function fetchAllPrices() {
  // Show loading state
  document.querySelectorAll('.watchlist-price').forEach(el => {
    el.style.opacity = '0.4';
  });

  const ASX_SYMBOLS = ['BHP', 'CBA', 'MQG', 'WDS'];
  const US_SYMBOLS = ['AAPL', 'TSLA', 'NVDA', 'MSFT', 'AMZN'];

  // Fetch all simultaneously
  const [cryptoData, ...stockResults] = await Promise.all([
    fetchCrypto(),
    ...ASX_SYMBOLS.map(s => 
      fetchPrice(s, true).then(r => r || fetchPriceFallback(s, true))
    ),
    ...US_SYMBOLS.map(s => 
      fetchPrice(s, false).then(r => r || fetchPriceFallback(s, false))
    )
  ]);

  // Update ASSETS object with live prices
  ASX_SYMBOLS.forEach((sym, i) => {
    if (stockResults[i] && stockResults[i].price > 0) {
      ASSETS[sym].price = stockResults[i].price;
      ASSETS[sym].change = stockResults[i].change;
      ASSETS[sym].high = stockResults[i].high;
      ASSETS[sym].low = stockResults[i].low;
      ASSETS[sym].volume = stockResults[i].volume;
    }
  });

  US_SYMBOLS.forEach((sym, i) => {
    const idx = i + ASX_SYMBOLS.length;
    if (stockResults[idx] && stockResults[idx].price > 0) {
      ASSETS[sym].price = stockResults[idx].price;
      ASSETS[sym].change = stockResults[idx].change;
      ASSETS[sym].high = stockResults[idx].high;
      ASSETS[sym].low = stockResults[idx].low;
    }
  });

  if (cryptoData) {
    if (cryptoData.BTC) ASSETS['BTC'].price = cryptoData.BTC.price;
    if (cryptoData.ETH) ASSETS['ETH'].price = cryptoData.ETH.price;
    if (cryptoData.SOL) ASSETS['SOL'].price = cryptoData.SOL.price;
  }

  // Update UI
  updateWatchlistPrices();
  updateTickerBar();
  updateChartCurrentPrice();
  updateTradeSummary();

  // Show live indicator
  document.querySelector('.live-indicator').textContent = 
    '● Live · refresh 15s';
  document.querySelector('.live-indicator').style.color = '#4ade80';

  // Restore opacity
  document.querySelectorAll('.watchlist-price').forEach(el => {
    el.style.opacity = '1';
  });
}

// Run on page load
fetchAllPrices();

// Refresh every 15 seconds
setInterval(fetchAllPrices, 15000);
